nop                             #do nothing

#arithmetics
add     dreg    sreg1   sreg2   #dreg.v = sreg1.v   +   sreg2.v
addi    dreg    sreg    imd     #dreg.v = sreg.v    +   imd
sub     dreg    sreg1   sreg2   #dreg.v = sreg1.v   -   sreg2.v
subi    dreg    sreg1   imd     #dreg.v = sreg1.v   -   imd
mul     dreg    sreg1   sreg2   #dreg.v = sreg1.v   *   sreg2.v
muli    dreg    sreg1   imd     #dreg.v = sreg1.v   *   imd
div     dreg    sreg1   sreg2   #dreg.v = sreg1.v   /   sreg2.v
divi    dreg    sreg1   imd     #dreg.v = sreg1.v   /   imd

#move, load, store
ld      dreg    sreg            #dreg.v = dmem[sreg.v]
ldi     dreg    imd             #dreg.v = dmem[imd]
st      dreg    sreg            #dmem[dreg.v] = sreg.v
sti     imd     sreg            #dmem[imd] = sreg.v

#branching
b       imd                     #pc   = imd
rb      imd                     #pc   = pc        +   imd
blt     imd     sreg1   sreg2   #pc   = sreg1.v   <   sreg2.v   ?   imd         : pc
rblt    imd     sreg1   sreg2   #pc   = sreg1.v   <   sreg2.v   ?   pc + imd    : pc
bz      imd     sreg1           #pc   = sreg1.v   ==  0         ?   imd         : pc
rbz     imd     sreg1           #pc   = sreg1.v   ==  0         ?   pc + imd    : pc

#control
halt                            #stop execution of program
